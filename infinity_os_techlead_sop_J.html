// HeartbeatSyncScheduler.java @Scheduled(cron = "0 0 2 * * ?") // 每天凌晨2点 public void syncRActual() { // 1. 从Hologres读昨天的汉询数据 String sql = """ SELECT distributor_id, sku_id, SUM(qty_sold) as total_qty FROM rixiao WHERE sale_date = CURRENT_DATE - 1 GROUP BY distributor_id, sku_id """; List rows = hologresJdbc.queryForList(sql); // 2. 写入 inf_heartbeat for (Map row : rows) { InfHeartbeat hb = new InfHeartbeat(); hb.setNodeId(row.get("distributor_id").toString()); hb.setSkuId(row.get("sku_id").toString()); hb.setRActual((Double) row.get("total_qty")); hb.setHeartbeatDate(LocalDate.now().minusDays(1)); heartbeatRepo.save(hb); } log.info("[ETL] r_actual同步完成：{}条", rows.size()); }
3
写库存同步任务
每天同步一次Dcloud库存数据到inf_heartbeat。
@Scheduled(cron = "0 30 2 * * ?") // 凌晨2:30 public void syncInventory() { String sql = """ SELECT distributor_id, sku_id, stock_qty FROM wm_stock_matnr WHERE update_time >= CURRENT_DATE - 1 """; List rows = hologresJdbc.queryForList(sql); for (Map row : rows) { // 更新 inf_heartbeat 的 inventory_units 字段 heartbeatRepo.updateInventory( row.get("distributor_id").toString(), row.get("sku_id").toString(), (Integer) row.get("stock_qty") ); } log.info("[ETL] 库存同步完成：{}条", rows.size()); }

1
运行数据质量检查SQL
-- 检查1：有没有 r_actual 为0的异常节点 SELECT node_id, COUNT(*) as zero_days FROM inf_heartbeat WHERE r_actual = 0 GROUP BY node_id HAVING zero_days > 3; -- 检查2：库存是否有异常高值（可能压货） SELECT node_id, inventory_units, r_actual, inventory_units/r_actual as water_level FROM inf_heartbeat WHERE heartbeat_date = CURDATE() - 1 ORDER BY water_level DESC LIMIT 20; -- 检查3：节点数量是否合理 SELECT COUNT(DISTINCT node_id) FROM inf_heartbeat; -- 应该接近实际经销商数量
2

// InventoryReportController.java @RestController @RequestMapping("/api/inventory") public class InventoryReportController { // SFA调用此接口推送库存数据 @PostMapping("/report") public ResponseEntity report(@RequestBody InventoryReportDTO dto, @RequestHeader("X-SFA-Token") String token) { // 验证token if (!tokenService.validate(token)) { return ResponseEntity.status(401).build(); } // 写入 inf_heartbeat InfHeartbeat hb = new InfHeartbeat(); hb.setNodeId(dto.getStoreId()); hb.setSkuId(dto.getSkuId()); hb.setInventoryShelf(dto.getShelfQty()); // 货架量 hb.setInventoryWarehouse(dto.getWarehouseQty()); // 仓库量 hb.setInventoryUnits(dto.getShelfQty() + dto.getWarehouseQty()); hb.setReportedBy(dto.getDsrId()); hb.setHeartbeatDate(LocalDate.now()); heartbeatRepo.save(hb); return ResponseEntity.ok(Map.of("status","ok","id",hb.getId())); } }
