# src/schedulers/etl_scheduler.py from apscheduler.schedulers.asyncio import AsyncIOScheduler from loguru import logger import psycopg2, pymysql from datetime import date, timedelta scheduler = AsyncIOScheduler(timezone="Asia/Shanghai") @scheduler.scheduled_job('cron', hour=2, minute=0) # 每天凌晨2:00 async def sync_r_actual(): """从汉询同步昨日销售数据 → inf_heartbeat.r_actual""" yesterday = date.today() - timedelta(days=1) logger.info(f"[ETL] 开始同步r_actual，日期：{yesterday}") # 1. 从Hologres读汉询数据 holo = psycopg2.connect(**HOLOGRES_CONFIG) cur = holo.cursor() cur.execute(""" SELECT distributor_id, sku_id, SUM(qty_sold) as r_actual FROM rixiao WHERE sale_date = %s GROUP BY distributor_id, sku_id """, (yesterday,)) rows = cur.fetchall() holo.close() # 2. 写入MySQL inf_heartbeat mysql = pymysql.connect(**MYSQL_CONFIG) cur2 = mysql.cursor() for dist_id, sku_id, r_actual in rows: cur2.execute(""" INSERT INTO inf_heartbeat (node_id, sku_id, r_actual, heartbeat_date) VALUES (%s, %s, %s, %s) ON DUPLICATE KEY UPDATE r_actual = VALUES(r_actual) """, (dist_id, sku_id, float(r_actual), yesterday)) mysql.commit() mysql.close() logger.info(f"[ETL] r_actual同步完成：{len(rows)}条") @scheduler.scheduled_job('cron', hour=2, minute=30) # 凌晨2:30 async def sync_inventory(): """从Dcloud同步库存数据 → inf_heartbeat.inventory_units""" logger.info("[ETL] 开始同步库存数据") holo = psycopg2.connect(**HOLOGRES_CONFIG) cur = holo.cursor() cur.execute(""" SELECT distributor_id, sku_id, stock_qty FROM wm_stock_matnr WHERE update_time >= CURRENT_DATE - 1 """) rows = cur.fetchall() holo.close() mysql = pymysql.connect(**MYSQL_CONFIG) cur2 = mysql.cursor() for dist_id, sku_id, stock_qty in rows: cur2.execute(""" UPDATE inf_heartbeat SET inventory_units = %s WHERE node_id = %s AND sku_id = %s AND heartbeat_date = CURDATE() - 1 """, (stock_qty, dist_id, sku_id)) mysql.commit() mysql.close() logger.info(f"[ETL] 库存同步完成：{len(rows)}条")

# src/services/wechat_service.py import httpx from loguru import logger from src.config import settings async def send_to_wechat(webhook_url: str, content: str) -> bool: async with httpx.AsyncClient() as client: resp = await client.post(webhook_url, json={ "msgtype": "text", "text": {"content": content} }) result = resp.json() if result.get("errcode") == 0: logger.info("[企微] 推送成功") return True logger.error(f"[企微] 推送失败：{result}") return False async def send_pressure_alert(node_name: str, water_level: float, suggestion: str): """推送A类压货预警""" content = f"""🦋【蝴蝶效应·A类压货预警】 经销商：{node_name} 当前水位：{water_level:.1f}a（正常2a满仓） 判断：库存过高，存在压货风险 建议行动：{suggestion} → 请城市经理48小时内处置 → 详情：https://infinity-os-sh.github.io/Infinity-os-SH/infinity_os_v30.html""" await send_to_wechat(settings.WECHAT_WEBHOOK_ALERT, content)

# src/agents/butterfly_effect.py from apscheduler.schedulers.asyncio import AsyncIOScheduler @scheduler.scheduled_job('cron', day_of_week='mon', hour=7, minute=0) async def run_butterfly_detection(): """每周一07:00：蝴蝶效应全链扫描（A值校准完成后）""" logger.info("[蝴蝶效应Agent] 开始全链扫描") mysql = pymysql.connect(**MYSQL_CONFIG) cur = mysql.cursor() cur.execute(""" SELECT h.node_id, n.node_name, h.inventory_units, h.r_actual, h.inventory_units / (h.r_actual * 30) as water_level FROM inf_heartbeat h JOIN inf_nodes n ON h.node_id = n.node_id WHERE h.heartbeat_date = CURDATE() - 1 """) rows = cur.fetchall() alert_count = 0 for node_id, node_name, inv, r_actual, water_level in rows: if water_level and water_level > 3.5: # A类：压货 # 写入预警表 cur.execute(""" INSERT INTO inf_alerts (node_id, alert_type, severity, details, created_at) VALUES (%s, 'A_PRESSURE', 'HIGH', %s, NOW()) """, (node_id, f"水位{water_level:.1f}a")) # 推企微 await send_pressure_alert( node_name, water_level, "建议协商促销消化库存，或暂停本周推单" ) alert_count += 1 mysql.commit() mysql.close() logger.info(f"[蝴蝶效应Agent] 扫描完成，发出{alert_count}条预警")

# src/agents/omakase_recommend.py @scheduler.scheduled_job('cron', day_of_week='mon', hour=8, minute=0) async def run_omakase(): """每周一08:00：无菜单推单（蝴蝶效应检测完成后）""" logger.info("[Omakase] 开始计算推单") # 获取无PAUSE状态的节点 mysql = pymysql.connect(**MYSQL_CONFIG) cur = mysql.cursor() cur.execute(""" SELECT h.node_id, n.node_name, t.r_target, h.inventory_units, t.grade FROM inf_heartbeat h JOIN inf_nodes n ON h.node_id = n.node_id JOIN inf_node_targets t ON h.node_id = t.node_id LEFT JOIN inf_alerts a ON h.node_id = a.node_id AND a.alert_type = 'OMAKASE_PAUSE' AND a.created_at > DATE_SUB(NOW(), INTERVAL 7 DAY) WHERE h.heartbeat_date = CURDATE() - 1 AND a.id IS NULL -- 没有PAUSE的才推单 """) for node_id, node_name, r_target, inv, grade in cur.fetchall(): a_value = r_target * 30 # a = r × 30天 target_level = 2.0 if grade <= 3 else 1.5 # 年级1-3目标2a，4-5目标1.5a target_inv = target_level * a_value order_qty = max(0, target_inv - inv) if order_qty > 0: # 生成推单建议 order_id = f"OMK-{date.today()}-{node_id}" msg = f"""🍣【Omakase推单建议】 经销商：{node_name} 当前库存：{inv:.0f}瓶（{inv/a_value:.1f}a） 目标水位：{target_level}a 建议补货：{order_qty:.0f}瓶 预计金额：¥{order_qty * 13:.0f}（按出厂价¥13/瓶估算） 推单编号：{order_id} 请在24小时内确认。 回复「确认-{order_id}」同意 回复「调整-{order_id}」修改""" await send_to_wechat(settings.WECHAT_WEBHOOK_ORDER, msg) # 写入inf_actions cur.execute(""" INSERT INTO inf_actions (node_id, action_type, details, status, created_at) VALUES (%s, 'OMAKASE_ORDER', %s, 'PENDING', NOW()) """, (node_id, f'{{"order_id":"{order_id}","qty":{order_qty}}}')) mysql.commit() mysql.close()

@scheduler.scheduled_job('cron', day_of_week='mon-fri', hour=6, minute=30) async def push_daily_tasks(): """工作日早上6:30：计算今日任务推给SFA""" mysql = pymysql.connect(**MYSQL_CONFIG) cur = mysql.cursor() # 找水位低或超期未访的门店 cur.execute(""" SELECT h.node_id, n.assigned_dsr, h.inventory_units / (h.r_actual * 7) as week_level, DATEDIFF(NOW(), n.last_visit_time) as days_since_visit, n.store_grade FROM inf_heartbeat h JOIN inf_nodes n ON h.node_id = n.node_id WHERE h.heartbeat_date = CURDATE() - 1 AND ( h.inventory_units / (h.r_actual * 30) < 1.5 -- 水位低 OR DATEDIFF(NOW(), n.last_visit_time) > 14 -- 超期未访 ) """) tasks = cur.fetchall() mysql.close() for node_id, dsr_id, level, days, grade in tasks: priority = "HIGH" if level < 1.0 or days > 21 else "NORMAL" note = f"水位{level:.1f}a偏低" if level < 1.5 else f"已{days}天未访" # 调用SFA任务派发API async with httpx.AsyncClient() as client: await client.post(f"{settings.SFA_API_BASE}/tasks", headers={"Authorization": f"Bearer {settings.SFA_API_KEY}"}, json={"dsr_id": dsr_id, "store_id": node_id, "task_type": "VISIT", "priority": priority, "note": note, "deadline": str(date.today())}) logger.info(f"[任务派发] {node_id} → DSR {dsr_id} | {note}")

@router.post("/api/tasks/complete") async def task_complete( task_id: str, store_id: str, dsr_id: str, task_type: str, note: str = "", x_sfa_token: str = Header(alias="X-SFA-Token") ): mysql = pymysql.connect(**MYSQL_CONFIG) cur = mysql.cursor() # 更新动作记录 cur.execute(""" UPDATE inf_actions SET status = 'COMPLETED', confirmed_at = NOW() WHERE node_id = %s AND action_type = %s AND status = 'DISPATCHED' """, (store_id, task_type)) # 写学习日志 cur.execute(""" INSERT INTO inf_learning_log (agent_name, action_id, human_decision, human_note, created_at) VALUES ('TASK_DISPATCH', %s, 'COMPLETED', %s, NOW()) """, (task_id, note)) mysql.commit() mysql.close() logger.info(f"[任务完成] {store_id} DSR:{dsr_id} 类型:{task_type}") return {"status": "recorded"}
