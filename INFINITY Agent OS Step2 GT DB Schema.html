---

设计要点

① transistor_signals 表是核心

  • 每次采集一条新记录,不覆盖,留时间序列
  • 10 个晶体管字段(s01–s10)严格按 DS② 字典
  • 有 has_conflict 标记多源冲突 → 喂 L8 学习
  • 有 ttl_expires_at 保鲜期 → 过期不入 L2 评估

② logic_gates 用 JSON 存规则

  • gate_logic 是 JSON,可在 SQL 外用规则引擎解析
  • 有 version 字段 → L8 学习可迭代新版门
  • 一个 (channel, stage) 对应一个 active 门

③ 严守 SKU 生命力底层

  • 所有表围绕"SKU 在某店活得怎么样"
  • 不是围绕"SR 做了什么"
  • 哪怕同一 SKU,在不同门店生命周期可不同(sku_lifecycle)

④ 字段命名规范

  • snake_case · 不缩写关键词
  • 时间戳:*_at · 计数:*_count · 标识:*_id
  • 业务字段都有 COMMENT(共 94 个)

⑤ 索引设计原则

  • 时间序列表:(sku_id, store_id, captured_at DESC) 高效拉最新
  • 主管/SR 视图:按 (assigned_to, status) 查待办
  • 共 22 个索引,覆盖核心查询路径

-- ============================================================
-- INFINITY Agent OS · Step 2 · 经营晶体管 DB 结构
-- 9 张表完整 DDL · MySQL 8.0
-- ============================================================
-- 设计依据:DS ⑪《经营晶体管数据库模型 V0.1》
-- 适用范围:GT 社区店第一个细胞 · 跑通后复制到其他业态
-- 工程纪律:严格按 DS V0.1 设计 · 不擅自加字段 · 不擅自改名
-- 署名 DS · 2026-05-29
-- ============================================================

SET NAMES utf8mb4;
SET FOREIGN_KEY_CHECKS = 0;

-- ============================================================
-- 表 1 / 9 · sku_products · SKU 产品表
-- ============================================================
-- 谁是 SKU · 产品的基础信息
DROP TABLE IF EXISTS sku_products;
CREATE TABLE sku_products (
  sku_id              VARCHAR(64)     NOT NULL COMMENT 'SKU ID·格式 BRAND_SKUNAME_SIZE·如 JYX_LIGHT_500',
  brand               VARCHAR(32)     NOT NULL COMMENT '品牌:六月鲜/味达美/禾然/葱伴侣/小康',
  product_name        VARCHAR(128)    NOT NULL COMMENT '产品名:六月鲜轻盐酱油 500ml',
  spec                VARCHAR(64)             COMMENT '规格:500ml/1.8L/小包装/家庭装',
  category            VARCHAR(64)             COMMENT '品类:酱油/料酒/醋',
  list_price          DECIMAL(10,2)           COMMENT '建议零售价(元)',
  gross_margin_pct    DECIMAL(5,4)            COMMENT '毛利率·0-1',
  status              VARCHAR(16)     NOT NULL DEFAULT 'active' COMMENT 'active/discontinued',
  created_at          TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at          TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (sku_id),
  INDEX idx_brand (brand),
  INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='SKU 产品主数据·DS⑪表1';


-- ============================================================
-- 表 2 / 9 · stores · 门店表
-- ============================================================
-- 经营场所·SKU 生命力的"栖息地"
DROP TABLE IF EXISTS stores;
CREATE TABLE stores (
  store_id            VARCHAR(64)     NOT NULL COMMENT '门店 ID·格式 CITY_CHANNEL_SUBTYPE_NUM·如 SH_GT_CVS_001',
  store_name          VARCHAR(128)    NOT NULL,
  channel_type        VARCHAR(32)     NOT NULL COMMENT '渠道类型·关联 channels 表',
  city                VARCHAR(32)     NOT NULL COMMENT '城市:上海/北京/广州',
  district            VARCHAR(64)             COMMENT '行政区:浦东/朝阳/天河',
  address             VARCHAR(256)            COMMENT '详细地址',
  lat                 DECIMAL(10,7)           COMMENT '纬度·用于 GPS 围栏校验',
  lng                 DECIMAL(10,7)           COMMENT '经度',
  store_size_sqm      INT                     COMMENT '店面平米数',
  owner_name          VARCHAR(64)             COMMENT '老板姓名·GT 社区店关键',
  owner_phone         VARCHAR(32)             COMMENT '老板电话',
  household_focus     TINYINT(1)              COMMENT '是否家庭客为主·GT 社区店判断',
  status              VARCHAR(16)     NOT NULL DEFAULT 'active',
  created_at          TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at          TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (store_id),
  INDEX idx_channel_city (channel_type, city),
  INDEX idx_city_district (city, district),
  INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='门店主数据·DS⑪表2';


-- ============================================================
-- 表 3 / 9 · channels · 渠道商业模式表
-- ============================================================
-- 每个渠道 = 一个"经营细胞模板"
DROP TABLE IF EXISTS channels;
CREATE TABLE channels (
  channel_type        VARCHAR(32)     NOT NULL COMMENT '渠道类型 ID:GT_CVS/MT/O2O/MEMBER/DISCOUNT/FOODSERVICE',
  channel_name        VARCHAR(64)     NOT NULL COMMENT '渠道中文名',
  business_model      TEXT                    COMMENT '商业模式描述·参考 DS④',
  primary_value       VARCHAR(128)            COMMENT '该渠道核心价值:高频复购/会员信任/即时需求',
  restock_cycle_days  INT                     COMMENT '典型补货周期·GT_CVS=7/14·MEMBER=30',
  custom_transistors  JSON                    COMMENT '业态专属晶体管列表·GT=[S09_owner_recommends]',
  status              VARCHAR(16)     NOT NULL DEFAULT 'active',
  created_at          TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (channel_type)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='渠道商业模式·DS⑪表3+DS④';


-- ============================================================
-- 表 4 / 9 · sku_lifecycle · SKU 生命周期表
-- ============================================================
-- 同一 SKU 在不同(店×渠道)的生命周期阶段·可不同
DROP TABLE IF EXISTS sku_lifecycle;
CREATE TABLE sku_lifecycle (
  lifecycle_id        BIGINT          NOT NULL AUTO_INCREMENT,
  sku_id              VARCHAR(64)     NOT NULL,
  store_id            VARCHAR(64)             COMMENT '可空·NULL=渠道城市级·非空=门店级',
  channel_type        VARCHAR(32)     NOT NULL,
  city                VARCHAR(32)     NOT NULL,
  lifecycle_stage     VARCHAR(16)     NOT NULL COMMENT 'intro导入/growth成长/mature成熟/refresh焕新/decline淘汰',
  entered_stage_at    TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '进入当前阶段时间',
  previous_stage      VARCHAR(16)             COMMENT '上一个阶段·用于轨迹追溯',
  updated_at          TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (lifecycle_id),
  UNIQUE KEY uk_sku_store_channel (sku_id, store_id, channel_type),
  INDEX idx_stage (lifecycle_stage),
  INDEX idx_sku_channel_city (sku_id, channel_type, city)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='SKU生命周期·DS⑪表4·L2门评判前必查';


-- ============================================================
-- 表 5 / 9 · transistor_signals · 经营晶体管信号表 【最重要】
-- ============================================================
-- 这是整个 L1 的核心 · 所有 L0 采集 → 这里 → L2 门读取
-- 每次采集一条记录 · 不覆盖 · 留时间序列
DROP TABLE IF EXISTS transistor_signals;
CREATE TABLE transistor_signals (
  signal_id           BIGINT          NOT NULL AUTO_INCREMENT,
  sku_id              VARCHAR(64)     NOT NULL,
  store_id            VARCHAR(64)     NOT NULL,
  channel_type        VARCHAR(32)     NOT NULL,
  captured_at         TIMESTAMP       NOT NULL COMMENT '信号产生时间·非入库时间',
  source              VARCHAR(32)     NOT NULL COMMENT 'sales_rep/pos/image_ai/manual',
  source_submission_id VARCHAR(64)            COMMENT '关联的 L0 提交 ID·可追溯',

  -- 10 个晶体管 · 严格按 DS② 字典定义
  s01_on_shelf        TINYINT(1)              COMMENT 'S01 是否上架·NULL=未采集',
  s02_in_stock        TINYINT(1)              COMMENT 'S02 是否有库存',
  s02_qty             INT                     COMMENT 'S02 库存数量(可选)',
  s03_main_shelf      TINYINT(1)              COMMENT 'S03 是否主货架',
  s04_second_display  TINYINT(1)              COMMENT 'S04 是否第二陈列',
  s04_display_type    VARCHAR(32)             COMMENT '第二陈列类型:dump_bin/endcap/checkout',
  s05_has_price_tag   TINYINT(1)              COMMENT 'S05 是否有价签',
  s05_shelf_price     DECIMAL(10,2)           COMMENT 'S05 货架价',
  s05_promo_price     DECIMAL(10,2)           COMMENT 'S05 促销价',
  s06_has_sellout     TINYINT(1)              COMMENT 'S06 是否动销·权威源=POS',
  s06_units_7d        INT                     COMMENT 'S06 7天销量',
  s06_txn_count_7d    INT                     COMMENT 'S06 7天交易笔数·防压货',
  s07_repurchase      TINYINT(1)              COMMENT 'S07 是否复购',
  s07_unique_buyers   INT                     COMMENT 'S07 去重买家数',
  s08_stockout        TINYINT(1)              COMMENT 'S08 是否断货',
  s08_stockout_days   INT                     COMMENT 'S08 断货天数',
  s08_root_cause      VARCHAR(64)             COMMENT 'S08 根因:distributor_oos/owner_not_ordered/...',
  s09_owner_recommends TINYINT(1)             COMMENT 'S09 老板是否推荐·GT业态专属',
  s09_recommend_level VARCHAR(16)             COMMENT 'proactive/if_asked/neutral/negative',
  s10_profit_target   TINYINT(1)              COMMENT 'S10 是否利润达标',
  s10_gross_margin    DECIMAL(5,4)            COMMENT 'S10 实际毛利率',

  -- 多源冲突标记·喂 L8 学习层
  has_conflict        TINYINT(1)      NOT NULL DEFAULT 0 COMMENT '与其他源冲突=1',
  conflict_notes      TEXT                    COMMENT '冲突详情·喂 L8 学习',

  -- 数据新鲜度·参考 Step1 L0→L1 mapping TTL 规则
  ttl_expires_at      TIMESTAMP               COMMENT '过期时间·过期后L2门视为未知',

  created_at          TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (signal_id),
  INDEX idx_sku_store_captured (sku_id, store_id, captured_at DESC),
  INDEX idx_captured_at (captured_at),
  INDEX idx_source (source),
  INDEX idx_conflict (has_conflict)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='【核心】经营晶体管信号·DS⑪表5·L0→L2 的桥梁';


-- ============================================================
-- 表 6 / 9 · logic_gates · 生命周期逻辑门定义表
-- ============================================================
-- L2 门的规则定义 · 每业态每阶段各一套
DROP TABLE IF EXISTS logic_gates;
CREATE TABLE logic_gates (
  gate_id             VARCHAR(64)     NOT NULL COMMENT 'GATE_<CHANNEL>_<STAGE>·如 GATE_GT_CVS_INTRO',
  channel_type        VARCHAR(32)     NOT NULL,
  lifecycle_stage     VARCHAR(16)     NOT NULL,
  gate_name           VARCHAR(128)    NOT NULL COMMENT '导入期生命信号门/成长门/...',
  gate_logic          JSON            NOT NULL COMMENT '门逻辑·JSON 表达·如 {"and":["s01","s02","s06"]}',
  pass_action         VARCHAR(256)            COMMENT '通过时的下一步·如 "进成长门"',
  fail_action         VARCHAR(256)            COMMENT '失败时下一步·如 "进L7行动建议"',
  description         TEXT                    COMMENT '业务说明·人类可读',
  version             VARCHAR(16)     NOT NULL DEFAULT 'v0.1' COMMENT '门版本·L8学习后可迭代',
  status              VARCHAR(16)     NOT NULL DEFAULT 'active',
  created_at          TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at          TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (gate_id),
  UNIQUE KEY uk_channel_stage_version (channel_type, lifecycle_stage, version),
  INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='生命周期逻辑门规则·DS⑪表6·L2引擎读取';


-- ============================================================
-- 表 7 / 9 · gate_results · 逻辑门判断结果表
-- ============================================================
-- L2 引擎每次评估的结果 · 历史可追溯
DROP TABLE IF EXISTS gate_results;
CREATE TABLE gate_results (
  result_id           BIGINT          NOT NULL AUTO_INCREMENT,
  signal_id           BIGINT          NOT NULL COMMENT '关联的 transistor_signals 记录',
  gate_id             VARCHAR(64)     NOT NULL,
  sku_id              VARCHAR(64)     NOT NULL,
  store_id            VARCHAR(64)     NOT NULL,
  channel_type        VARCHAR(32)     NOT NULL,
  lifecycle_stage     VARCHAR(16)     NOT NULL,
  result              VARCHAR(16)     NOT NULL COMMENT 'pass/fail/uncertain',
  reason              TEXT                    COMMENT '判断依据·如"S01✓ S02✓ 但 S06✗"',
  failed_signals      JSON                    COMMENT '失败的具体晶体管·如 ["s06"]',
  evaluated_at        TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (result_id),
  INDEX idx_sku_store_evaluated (sku_id, store_id, evaluated_at DESC),
  INDEX idx_result_stage (result, lifecycle_stage),
  INDEX idx_signal (signal_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='门判断结果·DS⑪表7·L3 渠道Agent读取';


-- ============================================================
-- 表 8 / 9 · agent_actions · Agent 行动建议表
-- ============================================================
-- L7 行动引擎输出 · 给一线人 / 主管的具体动作
DROP TABLE IF EXISTS agent_actions;
CREATE TABLE agent_actions (
  action_id           BIGINT          NOT NULL AUTO_INCREMENT,
  gate_result_id      BIGINT          NOT NULL COMMENT '关联的 gate_results',
  sku_id              VARCHAR(64)     NOT NULL,
  store_id            VARCHAR(64)     NOT NULL,
  action_type         VARCHAR(64)     NOT NULL COMMENT 'tasting/owner_training/get_second_display/promo/...',
  action_text         TEXT            NOT NULL COMMENT '人类可读·如"组织老板培训·攻克S09"',
  priority            VARCHAR(16)     NOT NULL DEFAULT 'normal' COMMENT 'high/normal/low',
  assigned_to         VARCHAR(64)             COMMENT '指派给谁·user_id',
  status              VARCHAR(16)     NOT NULL DEFAULT 'pending' COMMENT 'pending/in_progress/done/skipped',
  due_at              TIMESTAMP               COMMENT '截止时间',
  completed_at        TIMESTAMP               COMMENT '完成时间',
  completion_evidence TEXT                    COMMENT '完成证据·喂 L8 学习',
  effectiveness_score TINYINT                 COMMENT '事后效果评分 0-10·L8 学习的关键反馈',
  created_at          TIMESTAMP       NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (action_id),
  INDEX idx_assigned_status (assigned_to, status),
  INDEX idx_store_status (store_id, status),
  INDEX idx_gate_result (gate_result_id),
  INDEX idx_status_priority (status, priority)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Agent行动建议·DS⑪表8·L7输出';


-- ============================================================
-- 表 9 / 9 · resource_requests · 资源申请表
-- ============================================================
-- L8 资源调度入口 · Agent 触发的资源申请(陈列费/促销/培训等)
DROP TABLE IF EXISTS resource_requests;
CREATE TABLE resource_requests (
  request_id          BIGINT          NOT NULL AUTO_INCREMENT,
  action_id           BIGINT          NOT NULL COMMENT '关联的 agent_actions',
  sku_id              VARCHAR(64)     NOT NULL,
  store_id            VARCHAR(64)     NOT NULL,
  resource_type       VARCHAR(64)     NOT NULL COMMENT 'display_fee/promo_budget/training/supply_replenish/...',
