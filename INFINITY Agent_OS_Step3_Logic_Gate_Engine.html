引擎能做什么

1. 复合规则评估

JSON 规则 → Python 评估,支持:

# AND - 全部必须满足
{"rule": "AND", "conditions": ["s01_on_shelf", "s02_in_stock", "s06_has_sellout"]}

OR - 任意一个满足

{"rule": "OR", "conditions": ["s03_main_shelf", "s04_second_display"]}

NOT - 反转

{"rule": "NOT", "condition": "s08_stockout"}

嵌套

{"rule": "AND", "conditions": [
    "s01_on_shelf",
    {"rule": "OR", "conditions": ["s03_main_shelf", "s04_second_display"]}
]}

阈值比较

"s06_units_7d>=10"
"s10_gross_margin>0.3"
"s08_stockout=0"

2. 三态判断(关键诚实性)

  • ✅ PASS · 所有条件满足
  • ❌ FAIL · 至少一个条件明确失败
  • ⚠️ UNCERTAIN · 关键信号未采集 → 不假装判断
这是 Agent 诚实性的工程实现 —— "没采集到"不等于"是 false",Agent 不能把未知当失败,否则会推动错误的行动决策。

设计要点

① 规则用 JSON 存,不写死在代码里

  • logic_gates.gate_logic 字段是 JSON,业务可自助配置
  • L8 学习层未来可以自动迭代规则(版本化,A/B test)

② 三态判断保护 Agent 诚实

  • 字段为 None(未采集)→ uncertain · 不蒙
  • 字段明确 false → fail · 信号成立但不通过
  • 这两种 Agent 行动建议完全不同(补录 vs 行动)

③ 短路求值

  • OR 找到一个 PASS 就停,不必跑完所有条件
  • AND 找到一个明确 FAIL 也短路(性能 + 清晰)

④ 完全数据驱动

  • 加新门 = INSERT logic_gates · 不改代码
  • 改阈值 = UPDATE gate_logic JSON · 不改代码
  • 加新业态 = 新业态的 logic_gates 行 · 代码 0 改动

    class="tk">def evaluate(self, gate: LogicGate, signals: TransistorSignals) -> GateResult:

    class="tk">def _eval_rule(self, rule: Any, sig: TransistorSignals) -> dict:
        """
        返回 {
          passed: bool,
          has_uncertain: bool,
          failed: [signal names],
          uncertain: [signal names],
          checked: [signal names]  # 评估过的
        }

        # 复合规则
        if isinstance(rule, dict):
            r = rule.get("rule")
            if r == "AND":
                return self._eval_and(rule.get("conditions", []), sig)
            if r == "OR":
                return self._eval_or(rule.get("conditions", []), sig)
            if r == "NOT":
                return self._eval_not(rule.get("condition"), sig)
            # 字典但无 rule 字段·当成 AND
            if "conditions" in rule:
                return self._eval_and(rule["conditions"], sig)
            # 单个字段引用
            if "field" in rule:
                return self._eval_atom(rule["field"], sig)
        
        # 字符串单条件
        if isinstance(rule, str):
            return self._eval_atom(rule, sig)
        
        # 列表当 AND
        if isinstance(rule, list):
            return self._eval_and(rule, sig)
        
        # 兜底
        return self._empty_result(passed=False, failed=["INVALID_RULE"])

    class="tk">def _eval_and(self, conditions: list, sig: TransistorSignals) -> dict:
        """
        AND: 全部必须 pass · 任何一个 fail/uncertain 都影响最终结果
        
        关键诚实性: 字段为 None(未采集) → uncertain · 不是 fail
        因为"没采集到"≠"是 false"·Agent 不能把未知当失败
        """
        any_fail = False
        any_uncertain = False
        failed = []
        uncertain = []
        checked = []
        for cond in conditions:
            sub = self._eval_rule(cond, sig)
            checked.extend(sub["checked"])
            # 子结果有 uncertain·标记整体不确定
            if sub["uncertain"] or sub["has_uncertain"]:
                any_uncertain = True
                uncertain.extend(sub["uncertain"])
            # 子结果明确 failed(非未知导致)·标记 fail
            if sub["failed"]:
                any_fail = True
                failed.extend(sub["failed"])
            # 没失败也没不确定·但也没 pass(NOT 情况) · 不应发生
        
        # 优先级: 任何明确fail → fail · 否则有uncertain → uncertain · 否则 pass
        if any_fail:
            passed = False
            has_uncertain = False
        elif any_uncertain:
            passed = False  # uncertain 不算 pass
            has_uncertain = True
        else:
            passed = True
            has_uncertain = False
        
        return {
            "passed": passed,
            "has_uncertain": has_uncertain,
            "failed": failed,
            "uncertain": uncertain,
            "checked": checked,
        }

    class="tk">def _eval_or(self, conditions: list, sig: TransistorSignals) -> dict:
        """OR: 任何一个 pass 就 pass"""
        any_passed = False
        all_failed = []
        all_uncertain = []
        checked = []
        for cond in conditions:
            sub = self._eval_rule(cond, sig)
            checked.extend(sub["checked"])
            if sub["passed"]:
                any_passed = True
                break  # 短路求值
            all_failed.extend(sub["failed"])
            all_uncertain.extend(sub["uncertain"])
        return {
            "passed": any_passed,
            "has_uncertain": False if any_passed else bool(all_uncertain),
            "failed": [] if any_passed else all_failed,
            "uncertain": [] if any_passed else all_uncertain,
            "checked": checked,
        }

    class="tk">def _eval_not(self, condition: Any, sig: TransistorSignals) -> dict:
        """NOT: 反转结果"""
        if condition is None:
            return self._empty_result(passed=False, failed=["NOT_MISSING_CONDITION"])
        sub = self._eval_rule(condition, sig)
        return {
            "passed": not sub["passed"],
            "has_uncertain": sub["has_uncertain"],
            "failed": [] if not sub["passed"] else [f"NOT({s})" for s in sub["checked"]],
            "uncertain": sub["uncertain"],
            "checked": sub["checked"],
        }

    class="tk">def _eval_atom(self, atom: str, sig: TransistorSignals) -> dict:
        """
        评估一个原子条件:
        - "s01_on_shelf"       → 取字段·必须 truthy
        - "s06_units_7d>=10"   → 阈值比较
        - "s08_stockout=0"     → 等于
        """
        atom = atom.strip()
        
        # 阈值比较: 找操作符
        for op in (">=", "<=", "!=", "==", "=", ">", "<"):
            if op in atom:
                field_name, _, threshold = atom.partition(op)
                field_name = field_name.strip()
                threshold = threshold.strip()
                try:
                    threshold_val = float(threshold)
                except ValueError:
                    return self._empty_result(passed=False, failed=[f"BAD_THRESHOLD:{atom}"])
                
                actual = sig.get(field_name)
                if actual is None:
                    return {
                        "passed": False,
                        "has_uncertain": True,
                        "failed": [],
                        "uncertain": [field_name],
                        "checked": [field_name],
                    }
                # 布尔转 0/1
                if isinstance(actual, bool):
                    actual = 1 if actual else 0
                try:
                    actual_num = float(actual)
                except (TypeError, ValueError):
                    return self._empty_result(passed=False, failed=[f"NOT_NUMERIC:{field_name}"])
                
                passed = self._compare(actual_num, op, threshold_val)
                return {
                    "passed": passed,
                    "has_uncertain": False,
                    "failed": [] if passed else [f"{field_name}{op}{threshold} (actual={actual})"],
                    "uncertain": [],
                    "checked": [field_name],
                }
        
        # 简单字段引用: 必须 truthy
        field_name = atom
        value = sig.get(field_name)
        if value is None:
            return {
                "passed": False,
                "has_uncertain": True,
                "failed": [],
                "uncertain": [field_name],
                "checked": [field_name],
            }
        passed = bool(value)
        return {
            "passed": passed,
            "has_uncertain": False,
            "failed": [] if passed else [field_name],
            "uncertain": [],
            "checked": [field_name],
        }

    @staticmethod
    class="tk">def _compare(actual: float, op: str, threshold: float) -> bool:
        if op == ">=": return actual >= threshold
        if op == "<=": return actual <= threshold
        if op == ">":  return actual > threshold
        if op == "<":  return actual < threshold
        if op in ("==", "="): return actual == threshold
        if op == "!=": return actual != threshold
        return False

    @staticmethod
    class="tk">def _empty_result(passed: bool, failed: list = None) -> dict:
        return {
            "passed": passed, "has_uncertain": False,
            "failed": failed or [], "uncertain": [], "checked": [],

    class="tk">def _compose_reason(self, outcome: dict, sig: TransistorSignals) -> str:
        """生成给人看的判定原因文本"""
        if outcome["passed"] and not outcome["has_uncertain"]:
            return "✓ 所有条件满足 · 生命信号成立"
        if outcome["passed"] and outcome["has_uncertain"]:
            unc = ", ".join(outcome["uncertain"])
            return f"✓ 主要条件满足·但有不确定信号: {unc}"
        # fail
        parts = []
        if outcome["failed"]:
            parts.append("✗ 失败: " + ", ".join(outcome["failed"]))
        if outcome["uncertain"]:
            parts.append("⚠ 不确定: " + ", ".join(outcome["uncertain"]))
        return " | ".join(parts) or "✗ 评估失败"

        "captured_at": datetime.utcnow().isoformat() + "Z",

    cursor.execute(

    )

        (defaults["sku_id"], defaults["store_id"], defaults["channel_type"], stage)
    )
    conn.commit()
    conn.close()
    return signal_id
