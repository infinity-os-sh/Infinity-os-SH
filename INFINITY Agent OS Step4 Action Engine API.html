L7 五层结构(按 DS ③)

① ProblemIdentifier · 问题识别

从 gate_results 失败信号 → 分级问题(high/medium/low) 
# 示例: gate_result 含 ['s06_has_sellout'] →
Problem(problem_type="no_sellout", severity="high", ...)

② CauseReasoner · 原因推理

一个问题反推多个候选根因,各带 confidence: 
"无动销" 可能因为:
  · owner_no_recommend (0.85) · 老板不推荐·GT 关键
  · display_weak       (0.70) · 不在主货架
  · no_price_tag       (0.65) · 价签缺失
  · consumer_unknown   (0.50) · 消费者认知 (兜底)

③ LifecycleAssessor · 生命周期判断

判断是否触发阶段迁移: 
导入期 · 持续 3 次 high 失败 → 考虑淘汰(decline)
成长期 · 出现 high 失败       → 降回导入期
成熟期 · 任何失败             → 警示降级

④ ActionRecommender · 行动建议

基于(问题, 原因)→ 具体行动 + assigned_role + 目标晶体管 + 截止日: 
owner_no_recommend →
  · [high] owner_training "组织老板培训·攻克 S09"·assigned: GT_SR·3天
  · [normal] owner_visit "增加客情拜访"·assigned: GT_SR·7天

⑤ ResourceOrchestrator · 资源调度

判断行动是否需要资源 → 生成 resource_requests: 
promo            → promo_budget    ¥500  · DOM 审批
get_main_shelf   → display_fee     ¥1200 · DOM 审批
supply_replenish → supply_replenish 无金额 · DOM 审批

class ProblemIdentifier:
    """
    从 gate_results 中识别"真问题"
    
    不是所有 fail 都是真问题·要分级别:
    · S06 无动销 = 高严重(生命信号缺失·核心)
    · S03 不在主货架 = 中(可改善)
    · S04 无第二陈列 = 低(锦上添花)
    """

    # 失败晶体管 → 问题类型映射(可扩展)
    SIGNAL_TO_PROBLEM = {
        "s01_on_shelf":         ("not_listed",          "high"),
        "s02_in_stock":         ("stockout_current",    "high"),
        "s06_has_sellout":      ("no_sellout",          "high"),
        "s07_repurchase":       ("no_repurchase",       "medium"),
        "s08_stockout":         ("recent_stockout",     "high"),
        "s09_owner_recommends": ("owner_neutral",       "medium"),
        "s03_main_shelf":       ("not_main_shelf",      "medium"),
        "s04_second_display":   ("no_second_display",   "low"),
        "s05_has_price_tag":    ("price_tag_missing",   "medium"),
        "s10_profit_target":    ("profit_below_target", "medium"),
    }

    class="tk">def identify(self, gate_result_row: dict) -> list[Problem]:
        """从一行 gate_results 识别问题·一行可能含多个问题"""
        if gate_result_row["result"] == "pass":
            return []  # pass 无问题
        
        failed = json.loads(gate_result_row.get("failed_signals") or "[]")
        problems = []
        
        for signal_str in failed:
            # 处理 "s06_has_sellout=0 (actual=1)" 这种带表达式的
            signal_name = signal_str.split("=")[0].split(">")[0].split("<")[0].strip()
            
            mapping = self.SIGNAL_TO_PROBLEM.get(signal_name)
            if not mapping:
                continue
            problem_type, severity = mapping
            
            problems.append(Problem(
                sku_id=gate_result_row["sku_id"],
                store_id=gate_result_row["store_id"],
                channel_type=gate_result_row["channel_type"],
                lifecycle_stage=gate_result_row["lifecycle_stage"],
                severity=severity,
                problem_type=problem_type,
                failed_signals=[signal_name],
                gate_result_id=gate_result_row.get("result_id"),
                signal_id=gate_result_row.get("signal_id"),
            ))
        
        # 处理 uncertain(数据缺失)·也是问题但类型特殊
        if gate_result_row["result"] == "uncertain":
            problems.append(Problem(
                sku_id=gate_result_row["sku_id"],
                store_id=gate_result_row["store_id"],
                channel_type=gate_result_row["channel_type"],
                lifecycle_stage=gate_result_row["lifecycle_stage"],
                severity="medium",
                problem_type="data_missing",
                failed_signals=[],
                uncertain_signals=json.loads(gate_result_row.get("failed_signals") or "[]"),
                gate_result_id=gate_result_row.get("result_id"),
                signal_id=gate_result_row.get("signal_id"),
            ))
        
        return problems

class CauseReasoner:
    """
    从问题反推根因
    
    一个问题(如"无动销")可能有多个根因:
      - 消费者不知道 → 营销/认知
      - 价格太高 → 价格策略
      - 陈列差 → 看不到 → 物理动作
      - 老板不推 → 客情/培训
    
    输出多个候选 cause·各自带 confidence
    """

    class="tk">def reason(self, problem: Problem, signal_row: dict = None) -> list[Cause]:
        """
        基于问题类型 + 当前信号上下文·推理可能原因
        signal_row: 同时刻的完整晶体管信号·给推理提供更多上下文
        """
        causes = []
        sig = signal_row or {}
        
        if problem.problem_type == "no_sellout":
            # S06 无动销 · 多种可能根因
            
            # 根因 1: 没人推荐(S09) — 高置信
            if sig.get("s09_owner_recommends") == 0:
                causes.append(Cause(
                    cause_type="owner_no_recommend",
                    confidence=0.85,
                    evidence=["S09 老板不推荐 · GT 社区店关键"],
                    description="老板没有主动向消费者推荐·缺少&#x27;超级导购'力量"
                ))
            
            # 根因 2: 看不到(S03 主货架) — 中
            if sig.get("s03_main_shelf") == 0:
                causes.append(Cause(
                    cause_type="display_weak",
                    confidence=0.70,
                    evidence=["S03 不在主货架 · 消费者看不到"],
                    description="陈列位置弱 · 消费者注意力不到"
                ))
            
            # 根因 3: 价签缺(S05) — 中
            if sig.get("s05_has_price_tag") == 0:
                causes.append(Cause(
                    cause_type="no_price_tag",
                    confidence=0.65,
                    evidence=["S05 无价签 · 消费者犹豫"],
                    description="价签缺失 · 消费者不知价格不敢买"
                ))
            
            # 根因 4: 消费者认知 — 兜底(无具体证据时)
            if not causes:
                causes.append(Cause(
                    cause_type="consumer_unknown",
                    confidence=0.50,
                    evidence=["缺乏更具体的失败信号"],
                    description="可能是消费者认知问题 · 需要采集更多数据"
                ))
        
        elif problem.problem_type == "stockout_current" or problem.problem_type == "recent_stockout":
            # 断货 · 看 root_cause_guess 字段
            root = sig.get("s08_root_cause", "unknown")
            cause_map = {
                "distributor_oos": ("supply_chain", 0.90, "经销商无货 · 供应链问题"),
                "owner_not_ordered": ("owner_neglect", 0.85, "门店老板未下单 · 客情/培训问题"),
                "delivery_delay": ("logistics", 0.80, "配送延迟 · 物流问题"),
            }
            ct, cf, desc = cause_map.get(root, ("unknown", 0.40, "断货原因未知 · 需 SR 调查"))
            causes.append(Cause(
                cause_type=ct, confidence=cf,
                evidence=[f"S08 断货 · 根因猜测={root}"],
                description=desc
            ))
        
        elif problem.problem_type == "owner_neutral":
            causes.append(Cause(
                cause_type="owner_relationship_weak",
                confidence=0.75,
                evidence=["S09 老板态度 neutral"],
                description="老板对此 SKU 中立 · 需要客情+培训提升认同"
            ))
        
        elif problem.problem_type == "no_repurchase":
            causes.append(Cause(
                cause_type="product_experience",
                confidence=0.60,
                evidence=["S07 无复购"],
                description="消费者尝试后没回购 · 可能口感/价格/包装问题"
            ))
        
        elif problem.problem_type == "data_missing":
            causes.append(Cause(
                cause_type="data_gap",
                confidence=1.0,
                evidence=["关键信号未采集"],
                description="缺数据·Agent 无法判断·先补采集"
            ))
        
        # 兜底
        if not causes:
            causes.append(Cause(
                cause_type="generic",
                confidence=0.30,
                evidence=[f"problem_type={problem.problem_type}"],
                description="原因未明 · 需 SR 现场调研"
            ))
        
        return causes

# L7-③ 生命周期判断(Lifecycle Stage Assessment)
# ============================================================

class LifecycleAssessor:
    """
    根据问题严重程度 + 生命周期 · 判断是否触发阶段迁移
    
    例如:
      · 导入期持续 FAIL → 维持导入期·继续做小步动作
      · 成长期 FAIL → 降级回导入期(回炉重练)
      · 成熟期 FAIL → 降级回成长期(警示·避免成熟阶段衰退)
    """

    class="tk">def assess(self, problem: Problem, history_fail_count: int = 0) -> dict:
        """
        返回阶段迁移建议
        {
          "should_transition": bool,
          "new_stage": str or None,
          "reason": str
        }
        """
        stage = problem.lifecycle_stage
        sev = problem.severity
        
        # 导入期 · 严重失败累计 3 次 → 考虑淘汰
        if stage == "intro":
            if history_fail_count >= 3 and sev == "high":
                return {
                    "should_transition": True,
                    "new_stage": "decline",
                    "reason": "导入期持续 3 次以上 high 级失败 · 考虑淘汰"
                }
            return {"should_transition": False, "new_stage": None,
                    "reason": "导入期 fail 正常 · 维持继续培育"}
        
        # 成长期 · 严重 fail → 降回导入期
        if stage == "growth" and sev == "high":
            return {
                "should_transition": True,
                "new_stage": "intro",
                "reason": "成长期出现 high 级 fail · 降级回炉"
            }
        
        # 成熟期 · 任何 fail → 警示·降回成长期
        if stage == "mature" and sev in ("high", "medium"):
            return {
                "should_transition": True,
                "new_stage": "growth",
                "reason": "成熟期出现 fail · 警示性降级·防衰退"
            }
        
        return {"should_transition": False, "new_stage": None, "reason": "无需阶段迁移"}

        """基于问题 + 多个候选原因 · 生成行动建议(去重·按 priority 排)"""
        actions = []
        seen_types = set()
        
        for cause in causes:
            templates = self.CAUSE_TO_ACTIONS.get(cause.cause_type, [])
            for tpl in templates:
                if tpl["action_type"] in seen_types:
                    continue
                seen_types.add(tpl["action_type"])
                actions.append(ActionRecommendation(
                    action_type=tpl["action_type"],
                    action_text=tpl["text"],
                    priority=tpl["priority"],
                    assigned_role=tpl["role"],
                    target_signal=tpl["signal"],
                    due_in_days=tpl["days"],
                    expected_outcome=tpl["outcome"],
                ))
        
        # 按 priority 排:high → normal → low
        priority_order = {"high": 0, "normal": 1, "low": 2}
        actions.sort(key=lambda a: priority_order.get(a.priority, 99))
        return actions

    class="tk">def orchestrate(self, action: ActionRecommendation, problem: Problem) -> Optional[ResourceRequest]:
        """看这动作要不要资源·返回 None 表示不需要"""
        mapping = self.ACTION_TO_RESOURCE.get(action.action_type)

        return ResourceRequest(

                f"目标: {action.expected_outcome}"
            ),

        )

    class="tk">def process(self, gate_result_id: int) -> dict:
        """
        对一条 gate_results 执行 L7 完整 5 层流水
        返回完整推理链路·写库副作用
        """
        with self._conn() as c:
            row = c.execute(
                "SELECT * FROM gate_results WHERE result_id=?",
                (gate_result_id,)
            ).fetchone()
        
        if not row:
            return {"error": f"gate_result_id {gate_result_id} not found"}
        
        gate_result = dict(row)
        gate_result["result_id"] = gate_result_id
        
        # 取关联的信号(原因推理需要更多上下文)
        signal_row = {}
        if gate_result.get("signal_id"):
            with self._conn() as c:
                sr = c.execute(
                    "SELECT * FROM transistor_signals WHERE signal_id=?",
                    (gate_result["signal_id"],)
                ).fetchone()
                if sr:
                    signal_row = dict(sr)
        
        # 历史失败次数(给阶段判断用)
        history_fails = self._count_history_fails(
            gate_result["sku_id"], gate_result["store_id"]
        )
        
        # ─── L7 5 层推理 ───
        # ① 问题识别
        problems = self.identifier.identify(gate_result)
        if not problems:
            return {
                "gate_result_id": gate_result_id,
                "verdict": "pass·无需行动",
                "problems": [], "actions": [], "resources": []
            }
        
        all_actions = []
        all_resources = []
        all_causes_chain = []
        lifecycle_decisions = []
        
        for problem in problems:
            # ② 原因推理
            causes = self.reasoner.reason(problem, signal_row)
            # ③ 生命周期判断
            ls_decision = self.assessor.assess(problem, history_fails)
            # ④ 行动建议
            actions = self.recommender.recommend(problem, causes)
            
            # 落 actions 入库
            for action in actions:
                action_id = self._save_action(gate_result_id, problem, action)
                # ⑤ 资源调度
                resource = self.orchestrator.orchestrate(action, problem)
                if resource:
                    self._save_resource(action_id, problem, resource)
                    all_resources.append({**asdict(resource), "action_id": action_id})
                all_actions.append({**asdict(action), "action_id": action_id})
            
            all_causes_chain.append({
                "problem": asdict(problem),
                "causes": [asdict(c) for c in causes],
            })
            lifecycle_decisions.append(ls_decision)
        
        return {
            "gate_result_id": gate_result_id,
            "sku_id": gate_result["sku_id"],
            "store_id": gate_result["store_id"],
            "lifecycle_stage": gate_result["lifecycle_stage"],
            "verdict": gate_result["result"],
            "problems_count": len(problems),
            "actions_count": len(all_actions),
            "resources_count": len(all_resources),
            "reasoning_chain": all_causes_chain,
            "lifecycle_decisions": lifecycle_decisions,
            "actions": all_actions,
            "resources": all_resources,
        }

class="tk">def http_post(path: str, payload: dict) -> dict:
    req = urllib.request.Request(
        API_BASE + path,
        data=json.dumps(payload).encode("utf-8"),
        headers={"Content-Type": "application/json"},
        method="POST",
    )
    try:
        with urllib.request.urlopen(req, timeout=10) as resp:
            return json.loads(resp.read().decode("utf-8"))
    except urllib.error.HTTPError as e:
        return {"_status": e.code, "_error": e.read().decode("utf-8")}

    print("\n启动 Flask API server...")

    proc = subprocess.Popen(

        stdout=subprocess.PIPE, stderr=subprocess.PIPE,
    )
    
    try:
        if not wait_for_api(max_wait=10):
            print("❌ API server 未启动")
            stdout, stderr = proc.communicate(timeout=2)
            print("stdout:", stdout.decode()[:500])
            print("stderr:", stderr.decode()[:500])
            return False
        
        print("✅ API server 已起·端口 8001\n")
        
        ok = run_e2e()
        return ok
    finally:
        proc.terminate()
        try:
            proc.wait(timeout=3)
        except subprocess.TimeoutExpired:
            proc.kill()
        print("\n✅ API server 已关闭")


if __name__ == "__main__":
    ok = main()
