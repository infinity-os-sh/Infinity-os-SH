CREATE TABLE decision_log ( decision_id VARCHAR(32) PRIMARY KEY, -- DEC-{ts}-{role}-{seq} -- WHO 谁决策 decided_by VARCHAR(32) NOT NULL, role ENUM('CEO','GM','MCTL','CTL', 'DOM','OM','MOM','SR', 'DSR','ICA','其他'), org_scope VARCHAR(64), -- WHAT 做什么 decision_type ENUM('invest','restructure', 'staffing','pricing', 'channel','sku_lifecycle', 'promotion','authority'), decision_text TEXT NOT NULL, -- WHY 为什么 reason_text TEXT, trigger_event VARCHAR(128), -- 与 SKU 6 卡的关联 (核反应键) affected_skus JSON, -- [{sku_id, impact_type, impact_value}] affected_orgs JSON, -- [{org_id, impact}] affected_indvs JSON, -- [{individual_id, impact}] -- IMPACT 影响 (预期 + 实际) impact_predict JSON, -- {r_delta, a_delta, roi, time_horizon} impact_30d JSON, -- 30 天回填 (自动) impact_90d JSON, -- 90 天回填 (自动) -- AUTHORITY 权限 authority VARCHAR(32), approved_by VARCHAR(32), approved_at TIMESTAMP, -- 投入 budget DECIMAL(12,2), budget_source VARCHAR(32), budget_period VARCHAR(16), -- 时间戳 + 学习状态 created_at TIMESTAMP NOT NULL, verify_30d_at TIMESTAMP, verify_90d_at TIMESTAMP, learn_status ENUM('pending','verified', 'partial','failed'), learn_score DECIMAL(4,2), -- 实际/预期 比率 ai_lesson TEXT, -- AI 自动提取的规律 -- 索引 INDEX (decided_by, created_at), INDEX (role, decision_type), INDEX (learn_status), INDEX (verify_30d_at) -- 定时任务用 ); -- 关联表 · 决策与 SKU 多对多 CREATE TABLE decision_sku_link ( decision_id VARCHAR(32), sku_id VARCHAR(32), impact_type ENUM('positive','negative', 'neutral','mixed'), impact_value DECIMAL(12,2), -- ¥ 或 % 影响 PRIMARY KEY (decision_id, sku_id), FOREIGN KEY (decision_id) REFERENCES decision_log, FOREIGN KEY (sku_id) REFERENCES sku_6cards );
